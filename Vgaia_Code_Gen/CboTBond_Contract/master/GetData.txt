	/**
	 * GetData object from resultset <br>
	 *
	 * @param resultset object
	 * @return Object data
	 * @exception SQLException
	 */
	protected Object getData(ResultSet resultset)
		throws SQLException {

		// Declare CboTBond_Contract data
		CboTBond_ContractData data = new CboTBond_ContractData();
		
		data.setId(resultset.getBigDecimal(ID));
		data.setDelCd(resultset.getBigDecimal(DEL_CD));
		data.setContractNo(resultset.getString(CONTRACT_NO));
		data.setContractType(resultset.getBigDecimal(CONTRACT_TYPE));
		data.setBondCd(resultset.getString(BOND_CD));
		data.setSubAccoCd(resultset.getBigDecimal(SUB_ACCO_CD));
		data.setQty(resultset.getBigDecimal(QTY));
		data.setAmount(resultset.getBigDecimal(AMOUNT));
		data.setTaxAmt(resultset.getBigDecimal(TAX_AMT));
		data.setInterestAmt(resultset.getBigDecimal(INTEREST_AMT));
		data.setPaymentMethod(resultset.getBigDecimal(PAYMENT_METHOD));
		data.setIntPaymentMethod(resultset.getBigDecimal(INT_PAYMENT_METHOD));
		data.setContractDate(resultset.getBigDecimal(CONTRACT_DATE));
		data.setPaymentCashDate(resultset.getBigDecimal(PAYMENT_CASH_DATE));
		data.setPaymentSecDate(resultset.getBigDecimal(PAYMENT_SEC_DATE));
		data.setTransactionCd(resultset.getString(TRANSACTION_CD));
		data.setContractSts(resultset.getBigDecimal(CONTRACT_STS));
		data.setStatus(resultset.getBigDecimal(STATUS));
		data.setRemarks(resultset.getString(REMARKS));
		data.setRegDateTime(resultset.getTimestamp(REG_DATE_TIME));
		data.setRegUserId(resultset.getString(REG_USER_ID));
		data.setUpdDateTime(resultset.getTimestamp(UPD_DATE_TIME));
		data.setUpdUserId(resultset.getString(UPD_USER_ID));
		
		// Return data
		return data;
	}
